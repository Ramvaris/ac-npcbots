Rate.XP.BattlegroundKillWSG  = 1
Rate.XP.BattlegroundKillAB   = 1
Rate.XP.BattlegroundKillEOTS = 1
Rate.XP.BattlegroundKillSOTA = 1
Rate.XP.BattlegroundKillIC   = 1
